import javax.swing.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

public class Main {
    public static void main(String[] args) {
        String a = "Заполнить";
        String b = "Заполнить";
        String c = "Заполнить";
        new MyWindow(a,b,c);
    }
}


class MyWindow extends JFrame {
    public MyWindow(String a, String b, String c) {
        setTitle("MyWindows");
        setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
        setBounds(500, 300, 400, 400);


        setLayout(new BoxLayout(getContentPane(), BoxLayout.Y_AXIS));

        JTextField txt1 = new JTextField("Фамилия:"+a);
        txt1.setEnabled(false);
        JTextField txt2 = new JTextField("Имя:"+b);
        txt2.setEnabled(false);
        JTextField txt3 = new JTextField("Отчество:"+c);
        txt3.setEnabled(false);
        JButton btnOk = new JButton("Заполнить");

        txt1.setAlignmentY(CENTER_ALIGNMENT);
        txt2.setAlignmentY(CENTER_ALIGNMENT);
        txt3.setAlignmentY(CENTER_ALIGNMENT);
        btnOk.setAlignmentY(CENTER_ALIGNMENT);

        add(txt1);
        add(txt2);
        add(txt3);
        add(btnOk);
        setVisible(true);

        btnOk.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                new MyWindow2();
            }
        });
    }
}
